-- Cho 1 GuiObject kéo được tự do bằng vị trí UDim2.Offset.
-- Bị chặn khi dragLocked = true.
local function makeDraggable(guiObject, handle)
	handle = handle or guiObject
	local dragging = false
	local moved = false
	local dragStart, startPos
	local DRAG_THRESHOLD = 5

	handle.InputBegan:Connect(function(input)
		if dragLocked then return end
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			moved = false
			dragStart = input.Position
			startPos = guiObject.Position
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and not dragLocked and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
			local delta = input.Position - dragStart
			if not moved and delta.Magnitude > DRAG_THRESHOLD then
				moved = true
			end
			if moved then
				guiObject.Position = UDim2.new(
					startPos.X.Scale, startPos.X.Offset + delta.X,
					startPos.Y.Scale, startPos.Y.Offset + delta.Y
				)
			end
		end
	end)

	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = false
		end
	end)

	table.insert(allDraggables, guiObject)
end

local function setSpeedFromRatio(ratio)
	ratio = math.clamp(ratio, 0, 1)
	local speed = CONFIG.MinSpeed + (CONFIG.MaxSpeed - CONFIG.MinSpeed) * ratio
	CONFIG.FlySpeed = math.floor(speed + 0.5)
	speedLabel.Text = "SPEED: " .. CONFIG.FlySpeed
	updateSliderVisual(ratio)
end

-- set vị trí thanh trượt ban đầu khớp với FlySpeed mặc định

do
	local sliding = false

	local function ratioFromInput(inputPos)
		local trackPos = sliderTrack.AbsolutePosition.X
		local trackWidth = sliderTrack.AbsoluteSize.X
		if trackWidth <= 0 then return 0 end
		return (inputPos.X - trackPos) / trackWidth
	end

	sliderTrack.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
			sliding = true
			setSpeedFromRatio(ratioFromInput(input.Position))
		end
	end)

	sliderKnob.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
			sliding = true
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if sliding and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
			setSpeedFromRatio(ratioFromInput(input.Position))
		end
	end)

	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
			sliding = false
		end
	end)
end

local function bindHold(button, onStart, onEnd)
	button.MouseButton1Down:Connect(onStart)
	button.MouseButton1Up:Connect(onEnd)
	button.MouseLeave:Connect(onEnd)

	button.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch then

		if input.UserInputType == Enum.UserInputType.Touch then
			onEnd()
		end
	end)

-- SUPERMAN PROCEDURAL ANIMATION
----------------------------------------------------------------
local animTime = 0
local currentPitch   = 0  -- độ nghiêng người về trước (0..1)
local currentBank    = 0  -- độ nghiêng ngang trái/phải
local currentFloat   = 0  -- idle float phase
local savedMotorC0   = {} -- lưu C0 gốc của từng Motor6D để restore

local JOINTS = {
	"RootJoint", "Neck",
	"LeftShoulder", "RightShoulder",
	"LeftElbow",    "RightElbow",
	"LeftHip",      "RightHip",
	"LeftKnee",     "RightKnee",
}

local function lerp(a, b, t) return a + (b - a) * t end
local function lerpCF(a, b, t) return a:Lerp(b, t) end

local function getMotors(character)
	local motors = {}
	for _, name in ipairs(JOINTS) do
		local m = character:FindFirstChild(name, true)
		if m and m:IsA("Motor6D") then
			motors[name] = m
		end
	end
	return motors
end

local function saveMotors(character)
	savedMotorC0 = {}
	for _, name in ipairs(JOINTS) do
		local m = character:FindFirstChild(name, true)
		if m and m:IsA("Motor6D") then
			savedMotorC0[name] = m.C0
		end
	end
end

local function restoreMotors(character)
	for _, name in ipairs(JOINTS) do
		local m = character:FindFirstChild(name, true)
		if m and m:IsA("Motor6D") and savedMotorC0[name] then
			m.C0 = savedMotorC0[name]
		end
	end
end

local function applySupermanAnim(character, dt, forwardInput, rightInput, vertDir, speed)
	local motors = getMotors(character)
	animTime = animTime + dt

	-- Hệ số tốc độ (0..1) để scale độ mạnh animation theo tốc độ hiện tại
	local speedRatio = math.clamp(speed / 300, 0, 1)
	local moving = (math.abs(forwardInput) + math.abs(rightInput) + math.abs(vertDir)) > 0.05

	-- Pitch: càng nhanh / càng tiến về trước → càng chúi người xuống
	local targetPitch = 0
	if moving then
		targetPitch = (forwardInput * 0.7 + vertDir * 0.4) * speedRatio
		targetPitch = math.clamp(targetPitch, -0.8, 0.8)
	end
	currentPitch = lerp(currentPitch, targetPitch, dt * 5)

	-- Bank: nghiêng người sang trái/phải khi quẹo
	local targetBank = -rightInput * 0.35 * speedRatio
	currentBank = lerp(currentBank, targetBank, dt * 5)

	-- Float: nhấp nhô nhẹ khi idle (không di chuyển)
	local floatAmt = moving and 0 or 0.03
	currentFloat = lerp(currentFloat, floatAmt, dt * 3)
	local floatOffset = math.sin(animTime * 1.8) * currentFloat

	-- Gió thổi: torso rung nhẹ khi bay nhanh
	local windShake = speedRatio * 0.012
	local windX = math.sin(animTime * 7.3) * windShake
	local windZ = math.cos(animTime * 5.1) * windShake

	-- ── RootJoint ──
	-- Nghiêng toàn thân theo pitch và bank
	if motors.RootJoint and savedMotorC0.RootJoint then
		local base = savedMotorC0.RootJoint
		local pitchRot  = CFrame.Angles(-currentPitch * 1.1, 0, currentBank)
		local floatTrans = CFrame.new(0, floatOffset, 0)
		motors.RootJoint.C0 = lerpCF(motors.RootJoint.C0, base * floatTrans * pitchRot, dt * 10)
	end

	-- ── Neck ──
	-- Đầu ngẩng ngược lại với pitch để luôn nhìn về phía trước
	if motors.Neck and savedMotorC0.Neck then
		local base = savedMotorC0.Neck
		local neckCompensate = currentPitch * 0.7
		local neckSway = math.sin(animTime * 1.8) * 0.015 * speedRatio
		motors.Neck.C0 = lerpCF(motors.Neck.C0,
			base * CFrame.Angles(neckCompensate + windX, neckSway, -currentBank * 0.3),
			dt * 12)
	end

	-- ── Tay phải: giơ thẳng ra trước-lên (kiểu Superman iconic) ──
	if motors.RightShoulder and savedMotorC0.RightShoulder then
		local base = savedMotorC0.RightShoulder
		local extendAmt = lerp(0.1, 1.0, math.clamp(speedRatio * 1.5, 0, 1))
		-- Khi di chuyển: tay duỗi thẳng 45° lên và ra trước
		-- Khi idle: tay hơi xuôi tự nhiên, nhấp nhô theo float
		local targetRot
		if moving then
			targetRot = base * CFrame.Angles(
				-math.pi * 0.5 * extendAmt,   -- giơ lên
				0,
				-math.pi * 0.15 * extendAmt   -- xoay ra ngoài
			)
		else
			local idleSway = math.sin(animTime * 1.8 + 0.5) * 0.04
			targetRot = base * CFrame.Angles(0.1 + idleSway, 0, -0.15)
		end
		motors.RightShoulder.C0 = lerpCF(motors.RightShoulder.C0, targetRot, dt * 8)
	end

	-- ── Tay trái: duỗi ra sau-xuống, hơi choãi (balance) ──
	if motors.LeftShoulder and savedMotorC0.LeftShoulder then
		local base = savedMotorC0.LeftShoulder
		local extendAmt = lerp(0, 0.7, math.clamp(speedRatio * 1.2, 0, 1))
		local targetRot
		if moving then
			targetRot = base * CFrame.Angles(
				math.pi * 0.25 * extendAmt,   -- ra sau
				0,
				math.pi * 0.2 * extendAmt     -- choãi sang trái
			)
		else
			local idleSway = math.sin(animTime * 1.8) * 0.04
			targetRot = base * CFrame.Angles(0.05 + idleSway, 0, 0.15)
		end
		motors.LeftShoulder.C0 = lerpCF(motors.LeftShoulder.C0, targetRot, dt * 8)
	end

	-- ── Khuỷu tay: thẳng khi bay, hơi gập khi idle ──
	if motors.RightElbow and savedMotorC0.RightElbow then
		local base = savedMotorC0.RightElbow
		local bendAmt = moving and 0 or 0.25
		motors.RightElbow.C0 = lerpCF(motors.RightElbow.C0,
			base * CFrame.Angles(bendAmt, 0, 0), dt * 8)
	end
	if motors.LeftElbow and savedMotorC0.LeftElbow then
		local base = savedMotorC0.LeftElbow
		local bendAmt = moving and 0.08 or 0.3
		motors.LeftElbow.C0 = lerpCF(motors.LeftElbow.C0,
			base * CFrame.Angles(bendAmt, 0, 0), dt * 8)
	end

	-- ── Chân: duỗi thẳng ra sau, hơi chụm vào nhau ──
	if motors.RightHip and savedMotorC0.RightHip then
		local base = savedMotorC0.RightHip
		local extendAmt = lerp(0, 0.9, speedRatio)
		local kickBack = moving and (math.pi * 0.08 * extendAmt) or 0
		-- Chân hơi rung theo gió khi bay nhanh
		local legWind = math.sin(animTime * 6 + 1) * windShake * 0.5
		motors.RightHip.C0 = lerpCF(motors.RightHip.C0,
			base * CFrame.Angles(kickBack + legWind, 0, -0.05 * extendAmt),
			dt * 7)
	end
	if motors.LeftHip and savedMotorC0.LeftHip then
		local base = savedMotorC0.LeftHip
		local extendAmt = lerp(0, 0.9, speedRatio)
		local kickBack = moving and (math.pi * 0.08 * extendAmt) or 0
		local legWind = math.sin(animTime * 6) * windShake * 0.5
		motors.LeftHip.C0 = lerpCF(motors.LeftHip.C0,
			base * CFrame.Angles(kickBack + legWind, 0, 0.05 * extendAmt),
			dt * 7)
	end

	-- ── Đầu gối: thẳng hoàn toàn khi bay, hơi gập khi idle ──
	if motors.RightKnee and savedMotorC0.RightKnee then
		local base = savedMotorC0.RightKnee
		local bendAmt = moving and 0 or 0.12
		motors.RightKnee.C0 = lerpCF(motors.RightKnee.C0,
			base * CFrame.Angles(-bendAmt, 0, 0), dt * 7)
	end
	if motors.LeftKnee and savedMotorC0.LeftKnee then
		local base = savedMotorC0.LeftKnee
		local bendAmt = moving and 0 or 0.12
		motors.LeftKnee.C0 = lerpCF(motors.LeftKnee.C0,
			base * CFrame.Angles(-bendAmt, 0, 0), dt * 7)
	end
end

-- FLY ENGINE

local function startFly()
	local character = player.Character
	if not character then return end
	local hrp = character:FindFirstChild("HumanoidRootPart")
	local humanoid = character:FindFirstChildOfClass("Humanoid")
	if not hrp or not humanoid then return end

	flying = true
	humanoid.PlatformStand = true
	humanoid.AutoRotate = false

	saveMotors(character)
	animTime = 0
	currentPitch = 0
	currentBank  = 0
	currentFloat = 0

	bodyGyro = Instance.new("BodyGyro")
	bodyGyro.D = 100
	bodyGyro.P = 30000
	bodyGyro.MaxTorque = Vector3.new(400000, 400000, 400000)
	bodyGyro.CFrame = hrp.CFrame
	bodyGyro.Parent = hrp

	bodyVelocity = Instance.new("BodyVelocity")
	bodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
	bodyVelocity.Velocity = Vector3.new()
	bodyVelocity.Parent = hrp

	toggleBtn.Text = "FLY: ON"
end

local function stopFly()
	flying = false
	local character = player.Character
	if character then
		local humanoid = character:FindFirstChildOfClass("Humanoid")
		if humanoid then
			humanoid.PlatformStand = false
			humanoid.AutoRotate = true
		end
	end

	if bodyGyro then bodyGyro:Destroy() end
	if bodyVelocity then bodyVelocity:Destroy() end

	local ch = player.Character
	if ch then restoreMotors(ch) end

	toggleBtn.Text = "FLY: OFF"
end

local togglingFly = false
toggleBtn.MouseButton1Click:Connect(function()
	if togglingFly then return end
	togglingFly = true

	if flying then
		stopFly()
	else
		startFly()
	end

	task.delay(0.3, function()

RunService.RenderStepped:Connect(function(dt)
	if noclipping then
		local character = player.Character
		if character then
			for _, part in ipairs(character:GetDescendants()) do
				if part:IsA("BasePart") then
					part.CanCollide = false
				end
			end
		end
	end

	if not flying then return end

	local character = player.Character
	if not character then return end
	local hrp = character:FindFirstChild("HumanoidRootPart")
	local humanoid = character:FindFirstChildOfClass("Humanoid")
	if not hrp or not humanoid or not bodyVelocity or not bodyGyro then return end

	local moveDir = humanoid.MoveDirection

	local camCFrame = camera.CFrame
	local lookVector = camCFrame.LookVector
	local rightVector = camCFrame.RightVector

	local forwardInput = 0
	local rightInput = 0
	local flatLookForInput = Vector3.new(camCFrame.LookVector.X, 0, camCFrame.LookVector.Z)
	if flatLookForInput.Magnitude > 0 then
		flatLookForInput = flatLookForInput.Unit
		forwardInput = moveDir:Dot(flatLookForInput)
		local flatRight = Vector3.new(camCFrame.RightVector.X, 0, camCFrame.RightVector.Z).Unit
		rightInput = moveDir:Dot(flatRight)
	end

	local horizontal = (lookVector * forwardInput) + (rightVector * rightInput)
	local vertical = Vector3.new(0, verticalDirection, 0)

	local finalDir = horizontal + vertical
	if finalDir.Magnitude > 1 then
		finalDir = finalDir.Unit
	end

	bodyVelocity.Velocity = finalDir * CONFIG.FlySpeed

	if lookVector.Magnitude > 0 then
		bodyGyro.CFrame = CFrame.lookAt(hrp.Position, hrp.Position + lookVector)
	end

	applySupermanAnim(character, dt, forwardInput, rightInput, verticalDirection, CONFIG.FlySpeed)
end)
