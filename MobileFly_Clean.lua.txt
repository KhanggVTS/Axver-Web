-- ─── Config ───────────────────────────────────────────────────────────────────
local CFG = {
    baseSpeed      = 50,
    maxSpeed       = 100,
    boostMultipliers = { 1, 2.5, 4 },   -- Normal / Fast / Turbo
    boostIndex     = 1,
    pitchLerp      = 0.8,
    bankLerp       = 2.9,
    armFix         = true,              -- keep arms visible while flying
    defaultFOV     = 70,
    flyFOV         = 80,
    fovTween       = 0.4,
}

-- ─── Helpers ──────────────────────────────────────────────────────────────────
local function tween(obj, props, t, style, dir)
    style = style or Enum.EasingStyle.Quad
    dir   = dir   or Enum.EasingDirection.Out
    TweenService:Create(obj, TweenInfo.new(t, style, dir), props):Play()
end

local function getSpeed()
    return CFG.baseSpeed * CFG.boostMultipliers[CFG.boostIndex]
end

local function getCharacter()
    character = LocalPlayer.Character
    if not character then return false end
    humanoid  = character:FindFirstChildOfClass("Humanoid")
    hrp       = character:FindFirstChild("HumanoidRootPart")
    return humanoid and hrp
end

-- ─── Arm Visibility Fix ───────────────────────────────────────────────────────
local ARM_PARTS = {
    "Left Arm", "Right Arm",
    "LeftHand", "RightHand",
    "LeftLowerArm", "RightLowerArm",
    "LeftUpperArm", "RightUpperArm",
}

local function fixArms()
    if not CFG.armFix or not character then return end
    for _, name in ipairs(ARM_PARTS) do
        local part = character:FindFirstChild(name)
        if part then
            part.LocalTransparencyModifier = 0
            for _, child in ipairs(part:GetChildren()) do
                if child:IsA("Decal") or child:IsA("Texture") then
                    child.Transparency = 0
                elseif child:IsA("MeshPart") then
                    child.LocalTransparencyModifier = 0
                end
            end
        end
    end
end

-- ─── Fly Logic ────────────────────────────────────────────────────────────────
local function startFly()
    if not getCharacter() then return end

    humanoid.PlatformStand = true

    bodyVel = Instance.new("BodyVelocity")
    bodyVel.MaxForce   = Vector3.new(1e5, 1e5, 1e5)
    bodyVel.Velocity   = Vector3.zero
    bodyVel.Parent     = hrp

    bodyGyro = Instance.new("BodyGyro")
    bodyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
    bodyGyro.P         = 1e4
    bodyGyro.D         = 500
    bodyGyro.CFrame    = hrp.CFrame
    bodyGyro.Parent    = hrp

    tween(Camera, { FieldOfView = CFG.flyFOV }, CFG.fovTween)

    flyConn = RunService.RenderStepped:Connect(function(dt)
        if not (character and hrp) then return end

        -- Camera-relative input directions
        local camCF   = Camera.CFrame
        local forward = Vector3.new(camCF.LookVector.X, 0, camCF.LookVector.Z).Unit
        local right   = Vector3.new(camCF.RightVector.X, 0, camCF.RightVector.Z).Unit

        local move = Vector3.zero
        if UserInput:IsKeyDown(Enum.KeyCode.W) then move += forward end
        if UserInput:IsKeyDown(Enum.KeyCode.S) then move -= forward end
        if UserInput:IsKeyDown(Enum.KeyCode.D) then move += right   end
        if UserInput:IsKeyDown(Enum.KeyCode.A) then move -= right   end

        -- Vertical
        local vertical = 0
        if UserInput:IsKeyDown(Enum.KeyCode.Space)     then vertical =  1 end
        if UserInput:IsKeyDown(Enum.KeyCode.LeftShift) then vertical = -1 end

        local speed = getSpeed()
        local dir   = move ~= Vector3.zero and move.Unit or Vector3.zero

        bodyVel.Velocity = (dir * speed) + Vector3.new(0, vertical * speed * 0.6, 0)

        -- Gyro: face movement direction, pitch with camera
        if move ~= Vector3.zero then
            local targetCF = CFrame.lookAt(hrp.Position, hrp.Position + dir)
            local pitch    = math.clamp(camCF.LookVector.Y * -30, -45, 45)
            bodyGyro.CFrame = targetCF * CFrame.Angles(math.rad(pitch), 0, 0)
        else
            bodyGyro.CFrame = CFrame.new(hrp.Position) * CFrame.Angles(0, math.atan2(-hrp.CFrame.LookVector.X, -hrp.CFrame.LookVector.Z), 0)
        end

        if CFG.armFix then fixArms() end
    end)
end

local function stopFly()
    if flyConn then flyConn:Disconnect(); flyConn = nil end

    if bodyVel  then bodyVel:Destroy();  bodyVel  = nil end
    if bodyGyro then bodyGyro:Destroy(); bodyGyro = nil end

    if humanoid then humanoid.PlatformStand = false end

    tween(Camera, { FieldOfView = CFG.defaultFOV }, CFG.fovTween)
end

local function toggleFly()
    flying = not flying
    if flying then startFly() else stopFly() end
end

    if not draggingSpeed then return end
    if inp.UserInputType ~= Enum.UserInputType.Touch and inp.UserInputType ~= Enum.UserInputType.MouseMovement then return end

    local absPos = speedTrack.AbsolutePosition
    local absSize = speedTrack.AbsoluteSize
    local pos = inp.UserInputType == Enum.UserInputType.Touch
        and inp.Position.X
        or  inp.Position.X

    local ratio = math.clamp((pos - absPos.X) / absSize.X, 0, 1)
    CFG.baseSpeed = math.floor(10 + ratio * 290)
    speedVal.Text = tostring(CFG.baseSpeed)
    tween(speedFill, { Size = UDim2.new(ratio, 0, 1, 0) }, 0.05)

-- ── Panel open/close animation ───────────────────────────────────────────────
local panelOpen = false

local function openPanel()
    panel.Visible = true
    tween(panel, { Position = UDim2.new(1, -276, 0.5, -160) }, 0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
end

local function closePanel()
    tween(panel, { Position = UDim2.new(1, 20, 0.5, -160) }, 0.35, Enum.EasingStyle.Back, Enum.EasingDirection.In)
    task.delay(0.35, function()
        if not panelOpen then panel.Visible = false end
    end)
end

toggleBtn.Activated:Connect(function()
    panelOpen = not panelOpen
    if panelOpen then
        openPanel()
        tween(toggleBtn, { BackgroundColor3 = COLORS.accent }, 0.2)
    else
        closePanel()
        tween(toggleBtn, { BackgroundColor3 = COLORS.accentOff }, 0.2)
    end
end)

-- ── Fly toggle from panel ────────────────────────────────────────────────────
local function updateFlyUI()
    tween(statusDot,    { BackgroundColor3 = flying and COLORS.green or COLORS.red }, 0.2)
    tween(flyToggleBtn, { BackgroundColor3 = flying and COLORS.accent or COLORS.accentOff }, 0.2)
    statusText.Text  = flying and "Fly: ON" or "Fly: OFF"
    flyToggleBtn.Text = flying and "ON" or "OFF"
end

flyToggleBtn.Activated:Connect(function()
    toggleFly()
    updateFlyUI()
end)

-- Keyboard shortcut (PC): V to toggle fly, Q to cycle boost
UserInput.InputBegan:Connect(function(inp, gameProc)
    if gameProc then return end
    if inp.KeyCode == Enum.KeyCode.V then
        toggleFly()
        updateFlyUI()
    elseif inp.KeyCode == Enum.KeyCode.Q then
        CFG.boostIndex = (CFG.boostIndex % #CFG.boostMultipliers) + 1
        for j, b in ipairs(boostBtns) do
            tween(b, { BackgroundColor3 = j == CFG.boostIndex and COLORS.accent or COLORS.slider }, 0.2)
        end
    end
end)

-- ── Character respawn handling ───────────────────────────────────────────────
LocalPlayer.CharacterAdded:Connect(function(char)
    character = char
    humanoid  = char:WaitForChild("Humanoid")
    hrp       = char:WaitForChild("HumanoidRootPart")

    humanoid.Died:Connect(function()
        flying = false
        updateFlyUI()
        stopFly()
    end)

    if flying then
        task.wait(0.5)
        startFly()
    end
end)

if LocalPlayer.Character then
    character = LocalPlayer.Character
    humanoid  = character:FindFirstChildOfClass("Humanoid")
    hrp       = character:FindFirstChild("HumanoidRootPart")
    if humanoid then
        humanoid.Died:Connect(function()
            flying = false
            updateFlyUI()
            stopFly()
        end)
    end
end
